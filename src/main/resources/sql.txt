use bookstore;


### BOOK TABLE ###

DROP TABLE IF EXISTS book;
CREATE TABLE IF NOT EXISTS book
(
isbn bigint NOT NULL auto_increment,
category enum(
    'NONE',
    'ARTS_AND_MUSIC',
    'BIOGRAPHIES',
    'KIDS',
    'TECHNOLOGY',
    'COOKING',
    'EDUCATIONAL',
    'FICTION',
    'HEALTH_AND_FITNESS',
    'HISTORY',
    'HORROR',
    'LITERATURE',
    'ROMANCE',
    'SCIENCE_FICTION',
    'TRAVEL',
    'YOUNG_ADULT') NOT NULL,
title varchar(255) NOT NULL,
author varchar(255) NOT NULL,
price decimal(19,2) NOT NULL,
stock integer NOT NULL,
PRIMARY KEY (isbn)
)
engine=InnoDB
AUTO_INCREMENT = 1
DEFAULT CHARACTER SET = latin1;


### USER TABLE ###

DROP TABLE IF EXISTS users;
CREATE TABLE IF NOT EXISTS users
(
user_id bigint NOT NULL auto_increment,
first_name varchar(255) NOT NULL,
last_name varchar(255) NOT NULL,
address_line_1 varchar(255) NOT NULL,
address_line_2 varchar(255) NOT NULL,
post_code varchar(255) NOT NULL,
PRIMARY KEY (user_id)
)
engine=InnoDB
AUTO_INCREMENT = 1
DEFAULT CHARACTER SET = latin1;


### ORDER DETAILS TABLE ###

DROP TABLE IF EXISTS order_details;
CREATE TABLE IF NOT EXISTS order_details
(
order_details_id bigint NOT NULL auto_increment,
fk_user_id bigint NOT NULL,
total_price decimal NOT NULL,
order_date datetime NOT NULL,
PRIMARY KEY (order_details_id)
)
engine=InnoDB
AUTO_INCREMENT = 1
DEFAULT CHARACTER SET = latin1;

alter table order_details
add foreign key (fk_user_id) references users(user_id);


### ORDER ITEM TABLE ###

DROP TABLE IF EXISTS order_item;
CREATE TABLE IF NOT EXISTS order_item
(
order_item_id bigint NOT NULL auto_increment,
fk_order_details_id bigint NOT NULL,
fk_isbn bigint NOT NULL,
PRIMARY KEY (order_item_id)
)
engine=InnoDB
AUTO_INCREMENT = 1
DEFAULT CHARACTER SET = latin1;

alter table order_item
add foreign key (fk_order_details_id) references order_details(order_details_id);
alter table order_item
add foreign key (fk_isbn) references book(isbn);


### BASKET TABLE ###

DROP TABLE IF EXISTS basket;
CREATE TABLE IF NOT EXISTS basket
(
basket_id bigint NOT NULL auto_increment,
fk_user_id bigint NOT NULL,
total_price decimal NOT NULL,
PRIMARY KEY (basket_id)
)
engine=InnoDB
AUTO_INCREMENT = 1
DEFAULT CHARACTER SET = latin1;

alter table basket
add foreign key (fk_user_id) references users(user_id);


### BASKET ITEM TABLE ###

DROP TABLE IF EXISTS basket_item;
CREATE TABLE IF NOT EXISTS basket_item
(
basket_item_id bigint NOT NULL auto_increment,
fk_basket_id bigint NOT NULL,
fk_isbn bigint NOT NULL, #DEFAULT 0,
quantity integer NOT NULL,
PRIMARY KEY (basket_item_id)
)
engine=InnoDB
AUTO_INCREMENT = 1
DEFAULT CHARACTER SET = latin1;

alter table basket_item
add foreign key (fk_basket_id) references basket(basket_id);
alter table basket_item
add foreign key (fk_isbn) references book(isbn);
